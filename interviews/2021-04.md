function split(A) {
    let sum = A.reduce((a, b) => a + b)
    if(sum % 3 !== 0) return false
    let res = []
    for(let i = 0 ; i < A.length; i++) {
        let t = sum / 3
        while(t !== 0) {
            t -= A[i++]
        }
        res.push(--i)
    }
    return res
}

    { id: 1, value: 'test_1' },

    {

      children: [
        { id: 31, value: 'test_31' },

        {

          children: [
            { id: 331, value: 'test_331' },
            { id: 332, value: 'test_332' }
          ]
        }
      ]
    }
  ]
};
function f(obj) {
    let res = []
    let q = [[obj, -1]]
    
    while(q.length) {
        let [t, parentId] = q.shift()
        const { children } = t
        delete t.children
        t.parentId = parentId
        res.push(t)
        if(Array.isArray(children)) {
            children.forEach(child => {
                q.push([child, t.id])
            })
        }
    }
    return res
}
let res = f(obj)
