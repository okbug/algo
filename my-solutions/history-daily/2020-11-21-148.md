# 148. 排序链表

> https://leetcode-cn.com/problems/sort-list/

这一题和昨天的类似，如果作弊的话可以直接做出来。

```js
var sortList = function(head) {
    if(!head) return null
    let arr = [],p1 = head
    while(p1) {
        arr.push(p1.val)
        p1 = p1.next
    }
    arr = arr.sort((a, b) => (a - b))
    let newList = new ListNode(0)
    let p2 = newList;
    arr.forEach(i => {
        p2.next = new ListNode(i)
        p2 = p2.next
    })
    return newList.next
};
```
