# 耍赖做法

> https://leetcode-cn.com/problems/rotate-list

输入: 1->2->3->4->5->NULL, k = 2
输出: 4->5->1->2->3->NULL

```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var rotateRight = function(head, k) {
    if(!head) return null
    function ListNodeToArray(head) {
        let p = head
        let arr = []
        while(p&&p.next) {
            arr.push(p.val)
            p = p.next
        }
        arr.push(p.val)
        return arr
    }

    let arr = ListNodeToArray(head)
    if(k-arr.length > 10000) {
        k = k % arr.length
    }
    for(let i=0;i<k;i++) {
        arr.unshift(arr.pop())
    }
    function arrToListNode(arr) {
        let h = new ListNode(0)
        let p = h
        for(let i=0;i<arr.length;i++) {
            p.next = new ListNode(arr[i])
            p = p.next
        }
        return h.next
    }
    return arrToListNode(arr)
};
```



# 偷抄
```js
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @param {number} k
 * @return {ListNode}
 */
var rotateRight = function(head, k) {
  if (!head) return null;
  let curr = head;   // 一个指针
  let n = 0;
  while(++n && curr.next) {
    curr = curr.next;
  } // n是链表的长度
  curr.next = head;
  k = k % n; // 取模
  while(++k < n) {
    head = head.next; // 将倒数第k个元素变为头结点
  }
  let temp = head;
  head = head.next;
  temp.next = null; // 断开
  return head;
}
```

var reverseList = function(head) {
    let [prev, curr] = [null, head];
    while (curr) {
        let tmp = curr.next;    // 1. 临时存储当前指针后续内容
        curr.next = prev;       // 2. 反转链表
        prev = curr;            // 3. 接收反转结果
        curr = tmp;             // 4. 接回临时存储的后续内容
    }
    return prev;
};
