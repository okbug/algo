# 11. 盛最多水的容器

> https://leetcode-cn.com/problems/container-with-most-water/


```js
var maxArea = function(height) {
  let res = 0
  let i = 0, j = height.length - 1
  while(i < j) {
    const w = j - i
    const h = (height[i] > height[j]) ? height[j--] : height[i++]
    res = Math.max(res, w * h)
  }
  return res
};
```
