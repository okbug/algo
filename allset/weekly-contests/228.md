 * @return {number}

var minOperations = function(s) {
    let len = s.length
    let s1 = ''
    let s2 = ''
    for(let i = 0;i < len;i++) {
        if(i % 2 === 0) {
            s1 += '0'
            s2 += '1'
        } else {
            s1 += '1'
            s2 += '0'
        }
    }
    let diff1 = 0
    let diff2 = 0
    for(let i = 0;i < s.length;i++) {
        if(s[i] !== s1[i]) {
            diff1++
        }
        if(s[i] !== s2[i]) {
            diff2++
        }
    }
    return Math.min(diff1, diff2)
};
