[5685. 交替合并字符串](https://leetcode-cn.com/problems/merge-strings-alternately/)
```js
var mergeAlternately = function(w1, w2) {
    let a1 = w1.split('')
    let a2 = w2.split('')
    let a3 = []
    let len = Math.min(w1.length, w2.length)
    for(let i = 0;i < len;i++) {
        a3.push(a1.shift())
        a3.push(a2.shift())
    }
    a3 = [...a3,...a1,...a2]
    return a3.join('')
};
```
