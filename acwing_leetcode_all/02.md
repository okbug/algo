# 2. 两数相加

> https://leetcode-cn.com/problems/add-two-numbers/

```js
var addTwoNumbers = function(l1, l2) {
    let res = new ListNode(-1)
    let cur = res
    let carry = 0
    while(l1 || l2) {
        let n1 = l1 ? l1.val : 0
        let n2 = l2 ? l2.val : 0
        let sum = n1 + n2 + carry
        carry = sum >= 10 ? 1 : 0
        cur.next = new ListNode(sum % 10)
        cur = cur.next
        if(l1) l1 = l1.next
        if(l2) l2 = l2.next
    }
    if(carry) cur.next = new ListNode(1)
    return res.next
};
```
